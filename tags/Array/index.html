         Array.isArray(codData)?codData.map((item,index) => (
            <Card title={item.title} type="inner" id={index} key={index}>
              <div>

                {

                    return (
                    <Row>
                      {
                        values.map((itemValues, itemIndex) => {
                          const optionArr = itemValues.scope ? itemValues.scope.split(";") : [""];
                          return (
                            <span key={itemIndex}>
                              <Col span={4} style={{ marginBottom: 8 }}>
                                <div style={{ lineHeight: 2 }}>
                                  {itemValues.name} : &nbsp;
                                </div>
                              </Col>
                              <Col span={4} style={{ marginBottom: 8 }}>
                                {
                                  itemValues.type && itemValues.type === "下拉框" ? (
                                    <Select value={itemValues.value} style={{ width: 150 }} onChange={(record) => this.changeOptionValue(record,itemValues)}>
                                      {optionArr.map(optionKey => (
                                        <Option value={optionKey}>{optionKey}</Option>
                                      ))}
                                    </Select>
                                  ) : (
                                    <Input
                                      style={{ width: 150 }} 
                                      onChange={(e) => this.changeInputValue(e.target.value,itemValues)}
                                      defaultValue={itemValues.value}
                                    />
                                  )
                                }

                        })
                      }
                    </Row>//第二截止点
                    )
                  })
                }
                </Row>
